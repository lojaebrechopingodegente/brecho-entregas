⚠️ Execute este SQL no Editor do Supabase primeiro:

-- Tabela de controle das lives (código sequencial) create table if not exists lives_pg ( id uuid default gen_random_uuid() primary key, codigo text unique not null, -- Ex: LIVE-001, LIVE-002 data_live date not null, status text default 'ativa', -- ativa, finalizada total_vendas decimal(10,2) default 0, created_at timestamp with time zone default timezone('utc'::text, now()), updated_at timestamp with time zone default timezone('utc'::text, now()) ); -- Tabela de vendas vinculada ao código da live create table if not exists vendas_live ( id uuid default gen_random_uuid() primary key, codigo_live text not null references lives_pg(codigo), -- Chave estrangeira linha_id integer not null, data_live date not null, item integer, valor decimal(10,2) default 0, codigo text, comprador text, fila text, total decimal(10,2) default 0, created_at timestamp with time zone default timezone('utc'::text, now()), updated_at timestamp with time zone default timezone('utc'::text, now()), bloqueada boolean default false ); -- Índices para performance create index if not exists idx_vendas_live_codigo_live on vendas_live(codigo_live); create index if not exists idx_vendas_live_linha on vendas_live(linha_id); create index if not exists idx_vendas_live_data on vendas_live(data_live); create index if not exists idx_lives_pg_codigo on lives_pg(codigo);
