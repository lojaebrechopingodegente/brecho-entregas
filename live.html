🔧 Configuração do Supabase

⚠️ Execute este SQL no Editor do Supabase primeiro:

create table if not exists vendas_live ( id uuid default gen_random_uuid() primary key, linha_id integer not null, data_live date not null, item integer, valor decimal(10,2) default 0, codigo text, comprador text, fila text, total decimal(10,2) default 0, created_at timestamp with time zone default timezone('utc'::text, now()), updated_at timestamp with time zone default timezone('utc'::text, now()), bloqueada boolean default false ); create index if not exists idx_vendas_live_linha on vendas_live(linha_id); create index if not exists idx_vendas_live_data on vendas_live(data_live); create index if not exists idx_vendas_live_comprador on vendas_live(comprador);
