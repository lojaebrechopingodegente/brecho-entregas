🚀 Script PowerShell Completo

# ============================================
# Verificador de Entregas - PowerShell
# Busca tokens em PRODUÇÃO e ATIVOS
# ============================================

# Configurações do Supabase
$url = "SUA_URL_SUPABASE"
$apiKey = "SUA_API_KEY"

# Headers para requisição
$headers = @{
    "apikey" = $apiKey
    "Authorization" = "Bearer $apiKey"
    "Content-Type" = "application/json"
}

Write-Host "🔍 Buscando tokens de produção ativos..." -ForegroundColor Cyan
Write-Host ""

# Buscar token SuperFrete (PRODUÇÃO + ATIVO)
$urlSF = "$url/rest/v1/superfrete_config?select=api_token&environment=eq.production&is_active=eq.true&limit=1"
try {
    $responseSF = Invoke-RestMethod -Uri $urlSF -Headers $headers -Method Get
    if ($responseSF) {
        $tokenSF = $responseSF[0].api_token
        Write-Host "✅ Token SuperFrete PRODUÇÃO encontrado!" -ForegroundColor Green
    } else {
        Write-Host "❌ Token SuperFrete não encontrado (verifique se está salvo como produção+ativo)" -ForegroundColor Red
    }
} catch {
    Write-Host "❌ Erro ao buscar SuperFrete: $_" -ForegroundColor Red
}

# Buscar token Melhor Envio (PRODUÇÃO + ATIVO)
$urlME = "$url/rest/v1/melhor_envio_config?select=api_token&environment=eq.production&is_active=eq.true&limit=1"
try {
    $responseME = Invoke-RestMethod -Uri $urlME -Headers $headers -Method Get
    if ($responseME) {
        $tokenME = $responseME[0].api_token
        Write-Host "✅ Token Melhor Envio PRODUÇÃO encontrado!" -ForegroundColor Green
    } else {
        Write-Host "❌ Token Melhor Envio não encontrado (verifique se está salvo como produção+ativo)" -ForegroundColor Red
    }
} catch {
    Write-Host "❌ Erro ao buscar Melhor Envio: $_" -ForegroundColor Red
}

Write-Host ""
Write-Host "🚀 Iniciando verificação de entregas..." -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Yellow

# Aqui você pode adicionar a lógica de verificação das entregas
# usando os tokens $tokenSF e $tokenME

$resultado = @{
    data_verificacao = (Get-Date).ToString("yyyy-MM-dd HH:mm:ss")
    superfrete = @{
        token_encontrado = $tokenSF -ne $null
        ambiente = "production"
    }
    melhor_envio = @{
        token_encontrado = $tokenME -ne $null
        ambiente = "production"
    }
    entregas = @() # Preencher com dados reais das entregas
}

Write-Host ""
Write-Host "========================================" -ForegroundColor Yellow
Write-Host "📊 RESULTADO JSON (copie tudo entre as linhas amarelas):" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Yellow
$resultado | ConvertTo-Json -Depth 10
Write-Host "========================================" -ForegroundColor Yellow
💡 Dica: Substitua SUA_URL_SUPABASE e SUA_API_KEY pelos valores reais antes de executar.
